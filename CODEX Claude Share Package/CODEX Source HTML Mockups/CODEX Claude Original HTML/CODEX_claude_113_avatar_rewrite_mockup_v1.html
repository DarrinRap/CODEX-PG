Implementation sketch

class _PatientAvatar(QWidget):
    """Rounded-square patient photo, painted in paintEvent for DPR-safe rendering."""

    def __init__(self, size=36, radius=8, parent=None):
        super().__init__(parent)
        self._size = size
        self._radius = radius
        self._pixmap = None    # the pre-cropped square source pixmap
        self._is_panda = False # panda fallback flag
        self.setFixedSize(size, size)

    def set_photo(self, path):
        pm = QPixmap(path) if path and os.path.isfile(path) else QPixmap()
        if pm.isNull():
            self._pixmap = None
            self._is_panda = True
        else:
            # center-crop to square
            s = min(pm.width(), pm.height())
            x = (pm.width() - s) // 2
            y = (pm.height() - s) // 2
            self._pixmap = pm.copy(x, y, s, s)
            self._is_panda = False
        self.update()

    def paintEvent(self, event):
        p = QPainter(self)
        p.setRenderHint(QPainter.RenderHint.Antialiasing)
        p.setRenderHint(QPainter.RenderHint.SmoothPixmapTransform)

        # Clip to rounded square
        path = QPainterPath()
        path.addRoundedRect(0, 0, self._size, self._size, self._radius, self._radius)
        p.setClipPath(path)

        # Fill background
        p.fillRect(0, 0, self._size, self._size, QColor("#22223a"))

        if self._pixmap is not None:
            # drawPixmap scales source into widget rect in one pass
            p.drawPixmap(0, 0, self._size, self._size, self._pixmap)
        elif self._is_panda:
            p.setPen(QPen(QColor("#555")))
            p.setFont(QFont("Segoe UI", 16))
            p.drawText(QRectF(0, 0, self._size, self._size),
                       Qt.AlignmentFlag.AlignCenter, "\U0001F43C")
        p.end()

Key differences from the QLabel approach:
• No QPixmap created by us — Qt creates the target backing buffer via paintEvent, at correct DPR automatically.
• No setPixmap / QLabel alignment surprises — we own the paint.
• No border-radius QSS trick — real clip via QPainterPath.
• Pattern matches _CountBadge in same file, which renders correctly at 300% DPR.
• Single source-of-truth size: setFixedSize(36, 36) and paintEvent uses self._size.
